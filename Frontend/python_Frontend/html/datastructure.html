44. Add a method called `append` to the Linked List class that adds a new node to the end of the list.

Hint:
1. Define a method `append` inside the Linked List class to add a new node at the end.
2. To find the last node, iterate through the list until you find the node where `next` is None.
3. Update the `next` of the last node to point to the new node. 

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def append(self, data):
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            return
        last = self.head
        while last.next:
            last = last.next
        last.next = new_node

# Create a linked list and add a node
linked_list = LinkedList()
linked_list.append(10)
linked_list.append(20)

# Print the first node's data
print(linked_list.head.data)  # Output will be 10

45. Add a method called `traverse` to the Linked List class that prints the
data of each node.

Hint:
1. Define a method `traverse` that starts from the head node.
2. Use a while loop to go through each node until you reach a node where `next` is None.
3. Print the `data` of each node as you traverse. 

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def append(self, data):
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            return
        last = self.head
        while last.next:
            last = last.next
        last.next = new_node

    def traverse(self):
        current = self.head
        while current:
            print(current.data)
            current = current.next

# Create a linked list, add nodes and traverse
linked_list = LinkedList()
linked_list.append(10)
linked_list.append(20)
linked_list.append(30)

linked_list.traverse()  # Output will be 10, 20, 30

1. Define a method `reverse` that does not take any arguments.
2. Traverse the list while keeping track of the previous, current, and next nodes.
3. Update the `next` pointer of each node to point to the previous node.
4. Set the head of the list to the last node at the end of the traversal. 

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

    def append(self, data):
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            return
        last = self.head
        while last.next:
            last = last.next
        last.next = new_node

    def reverse(self):
        prev = None
        current = self.head
        while current:
            next_node = current.next
            current.next = prev
            prev = current
            current = next_node
        self.head = prev

    def traverse(self):
        current = self.head
        while current:
            print(current.data, end=" ")
            current = current.next
        print()

76. Add a method called extract_min to the Min Heap class that removes and returns the minimum element from the heap. Create an object, insert elements, and extract the minimum element.

class MinHeap:
    def __init__(self):
        self.heap = []

    def insert(self, val):
        self.heap.append(val)
        self.heapify_up(len(self.heap) - 1)
    
    def heapify_up(self, index):
        while index > 0 and self.heap[index] < self.heap[(index - 1) // 2]:
            self.heap[index], self.heap[(index - 1) // 2] = self.heap[(index - 1) // 2], self.heap[index]
            index = (index - 1) // 2

    def extract_min(self):
        if len(self.heap) > 1:
            self.heap[0], self.heap[-1] = self.heap[-1], self.heap[0]
            min_val = self.heap.pop()
            self.heapify_down(0)
            return min_val
        return self.heap.pop()

    def heapify_down(self, index):
        left = 2 * index + 1
        right = 2 * index + 2
        smallest = index

        if left < len(self.heap) and self.heap[left] < self.heap[smallest]:
            smallest = left
        if right < len(self.heap) and self.heap[right] < self.heap[smallest]:
            smallest = right
        if smallest != index:
            self.heap[index], self.heap[smallest] = self.heap[smallest], self.heap[index]
            self.heapify_down(smallest)

heap1.insert(20)
heap1.insert(5)

# Extract the minimum element
min_val = heap1.extract_min()

class TrieNode:
    def __init__(self):
        self.children = {}
        self.end_of_word = False

class Trie:
    def __init__(self):
        self.root = TrieNode()
    
    def insert(self, word):
        node = self.root
        for char in word:
            if char not in node.children:
                node.children[char] = TrieNode()
            node = node.children[char]
        node.end_of_word = True
    
    def print_trie(self, node=None, prefix=""):
        if node is None:
            node = self.root
        if node.end_of_word:
            print(prefix)
        for char, child in node.children.items():
            self.print_trie(child, prefix + char)

class HashTable:
    def __init__(self, size):
        self.size = size
        self.table = [None] * size

        return hash(key) % self.size

82. Implement the Insertion Sort algorithm. Sort the list [12, 11, 13, 5, 6] using Insertion Sort and print the sorted list.

Hint:
1. Start by iterating over the list from the second element.
2. For each element, compare it to the elements before it and shift them to the right if needed.
3. Insert the current element into the correct position.
4. Print the list after sorting. 

def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

# Example list
arr = [12, 11, 13, 5, 6]
sorted_arr = insertion_sort(arr)
print(sorted_arr)

83. Implement the Bubble Sort algorithm. Sort the list [64, 34, 25, 12, 22, 11, 90] using Bubble Sort and print the sorted list.

Hint:
1. Start by iterating through the list repeatedly.
2. For each iteration, compare adjacent elements and swap them if they are in the wrong order.
3. After each pass, the largest element will bubble to the end of the list.
4. Print the list after sorting. 

def bubble_sort(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
    return arr

# Example list

print(sorted_arr)

84. Implement the Selection Sort algorithm. Sort the list [29, 10, 14, 37, 13] using Selection Sort and print the sorted list.

Hint:
1. Start by finding the minimum element in the list.
2. Swap the minimum element with the first unsorted element.
3. Repeat the process for the remaining unsorted portion.
4. Print the list after sorting. 

def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

# Example list
arr = [29, 10, 14, 37, 13]
sorted_arr = selection_sort(arr)
print(sorted_arr)

85. Implement the Merge Sort algorithm. Sort the list [38, 27, 43, 3, 9, 82, 10] using Merge Sort and print the sorted list.

Hint:
1. Divide the list into two halves recursively until each sublist has one element.
2. Merge the sublists back together in sorted order.
3. Print the list after sorting. 

def merge_sort(arr):
    if len(arr) > 1:
        mid = len(arr) // 2
        left_half = arr[:mid]
        right_half = arr[mid:]

        merge_sort(left_half)
        merge_sort(right_half)

        i = j = k = 0
        while i < len(left_half) and j < len(right_half):
            if left_half[i] < right_half[j]:
                arr[k] = left_half[i]
                i += 1
            else:
                arr[k] = right_half[j]
                j += 1
            k += 1

        while i < len(left_half):
            arr[k] = left_half[i]
            i += 1
            k += 1

        while j < len(right_half):
            arr[k] = right_half[j]
            j += 1
            k += 1
    return arr

# Example list
arr = [38, 27, 43, 3, 9, 82, 10]
sorted_arr = merge_sort(arr)
print(sorted_arr)
